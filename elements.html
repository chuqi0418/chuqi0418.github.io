代码示例（牛顿迭代法）

def newton_sqrt(x, tolerance=1e-10):
    guess = x / 2
    while True:
        new_guess = 0.5 * (guess + x / guess)
        if abs(new_guess - guess) < tolerance:
            return new_guess
        guess = new_guess
